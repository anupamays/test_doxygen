>>> infer_compression('http://foo/bar.tgz')

>>> infer_compression('file.bz')

>>> infer_compression('file.xz')

setuptools._vendor.jaraco.context.temp_dir ( remover = shutil.rmtree)

Create a temporary directory context.

Pass a custom remover to override the removal behavior.

>>> import pathlib
>>> with temp_dir() as the_dir:
... assert os.path.isdir(the_dir)
... _ = pathlib.Path(the_dir).joinpath('somefile').write_text('contents')
>>> assert not os.path.exists(the_dir)
